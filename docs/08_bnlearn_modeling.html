get_diff <- function(x) {
  c(NA, diff(x))
}

data_delta <- data %>% select(
  hosp_id, 
  measure_point,
  age,
  starts_with("OCI"),
  -OCI_total,
  EPS:miscellany, RH,
  starts_with('PANSS'),
  -PANSS_total,
  starts_with('HP'),
  -HP_total) %>% 
  group_by(hosp_id) %>% 
  arrange(measure_point) %>% 
  mutate_at(vars(-hosp_id, -measure_point, -age), get_diff) %>% 
  ungroup() %>% drop_na
net2<- data_delta %>%
